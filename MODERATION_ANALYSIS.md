private function calculateTrustScore(int $userId): float {
    $approved = Ad::where('user_id', $userId)
        ->where('status', AdStatus::ACTIVE)->count();
    $rejected = Ad::where('user_id', $userId)
        ->where('status', AdStatus::REJECTED)->count();
